// استبدل دوال الحفظ والجلب السابقة بهذه الأكواد async function saveInvoice() { const inv = { name: document.getElementById('custName').value, date: document.getElementById('invDate').value, amount: document.getElementById('invAmount').value }; const response = await fetch('/api/invoices', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(inv) }); if(response.ok) { alert("تم الحفظ في قاعدة البيانات بنجاح"); showSection('archive'); } } async function renderTable() { const response = await fetch('/api/invoices'); const data = await response.json(); const body = document.getElementById('invoiceBody'); body.innerHTML = data.map(i => ` ${i.customer_name} ${new Date(i.invoice_date).toLocaleDateString('ar-SA')} ${i.amount} ر.س `).join(''); }
